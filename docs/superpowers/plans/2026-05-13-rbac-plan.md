# RBAC 权限系统 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an RBAC permission system to control which pages and menu items are visible to each user based on their assigned role.

**Architecture:** Standard 4-table RBAC (role, user_role, permission, role_permission). Permissions are loaded at login time and written into session. Razor Pages check session permissions — sidebar menus are conditionally rendered, and pages without permission redirect to Dashboard with a toast message.

**Tech Stack:** .NET 10.0, ASP.NET Core Razor Pages, EF Core 10.0.7 (EnsureCreated), session-based auth, Bootstrap 5.

### Task 3: Update Login Flow — Load Permissions into Session

**Files:**
- Modify: `Pages/Login.cshtml.cs`

- [ ] **Step 1: Update `OnPost` to load permissions**

After successful password validation, query permissions and write to session:

```csharp
public async Task<IActionResult> OnPost()
{
    try
    {
        var user = await _db.TbUsers.FirstOrDefaultAsync(u => u.Username == Username);

        if (user != null && user.Password == Password)
        {
            // Load roles
            var roleKeys = await (
                from ur in _db.UserRoles
                join r in _db.Roles on ur.RoleId equals r.Id
                where ur.UserId == user.Id
                select r.RoleKey
            ).ToListAsync();

            // Load permissions (through role -> permission join)
            var permissionKeys = await (
                from rp in _db.RolePermissions
                join p in _db.Permissions on rp.PermissionId equals p.Id
                join ur in _db.UserRoles on rp.RoleId equals ur.RoleId
                where ur.UserId == user.Id
                select p.PermissionKey
            ).ToListAsync();

            HttpContext.Session.SetString("UserId", user.Id.ToString());
            HttpContext.Session.SetString("UserName", user.Username);
            HttpContext.Session.SetString("RealName", user.RealName);
            HttpContext.Session.SetString("Gender", user.Gender);
            HttpContext.Session.SetString("UserRoles", string.Join(",", roleKeys));
            HttpContext.Session.SetString("UserPermissions", string.Join(",", permissionKeys));

            return RedirectToPage("/Dashboard");
        }

        ErrorMessage = "用户名或密码错误";
    }
    catch (Exception)
    {
        ErrorMessage = "服务异常，请稍后重试";
    }

    return Page();
}
```

```csharp
using Microsoft.AspNetCore.Mvc;
using Microsoft.AspNetCore.Mvc.RazorPages;

namespace xiaoliran.Models
{
    public abstract class PermissionPageModel : PageModel
    {
        protected abstract string PermissionKey { get; }

        public override async Task OnPageHandlerExecutionAsync(PageHandlerExecutingContext executing, PageHandlerExecutionDelegate next)
        {
            var session = HttpContext.Session;
            var userId = session.GetString("UserId");

            if (string.IsNullOrEmpty(userId))
            {
                HttpContext.Response.Redirect("/Login");
                return;
            }

            var permissions = session.GetString("UserPermissions") ?? "";
            if (!permissions.Contains(PermissionKey))
            {
                HttpContext.Response.Redirect("/Dashboard?toast=无权限访问此页面");
                return;
            }

            await next();
        }
    }
}
```

{
    if (!ModelState.IsValid)
    {
        Message = "请检查输入是否完整";

        return Page();

Add the `SeedInitialData` method before `app.Run()`:

```csharp
void SeedInitialData(AppDbContext db)
{
    if (db.Roles.Any()) return; // Already seeded

    var adminRole = new Role
    {
        RoleKey = "admin",
        RoleName = "管理员",
        Description = "拥有全部权限"
    };

    var userRole = new Role
    {
        RoleKey = "user",
        RoleName = "普通用户",
        Description = "基础查看权限"
    };

    db.Roles.AddRange(adminRole, userRole);
    db.SaveChanges();

    var dashboardPerm = new Permission
    {
        PermissionKey = "view_dashboard",
        PermissionName = "查看首页",
        Module = "system"
    };

    var laundryPerm = new Permission
    {
        PermissionKey = "view_laundryshop",
        PermissionName = "查看洗衣店管理",
        Module = "laundry"
    };

    var clothingPerm = new Permission
    {
        PermissionKey = "view_clothing",
        PermissionName = "查看衣服管理",
        Module = "clothing"
    };

    db.Permissions.AddRange(dashboardPerm, laundryPerm, clothingPerm);
    db.SaveChanges();

    // Admin gets all permissions
    db.RolePermissions.AddRange(
        new RolePermission { RoleId = adminRole.Id, PermissionId = dashboardPerm.Id },
        new RolePermission { RoleId = adminRole.Id, PermissionId = laundryPerm.Id },
        new RolePermission { RoleId = adminRole.Id, PermissionId = clothingPerm.Id }
    );

    // User gets dashboard and clothing
    db.RolePermissions.AddRange(
        new RolePermission { RoleId = userRole.Id, PermissionId = dashboardPerm.Id },
        new RolePermission { RoleId = userRole.Id, PermissionId = clothingPerm.Id }
    );

    db.SaveChanges();
}
```
